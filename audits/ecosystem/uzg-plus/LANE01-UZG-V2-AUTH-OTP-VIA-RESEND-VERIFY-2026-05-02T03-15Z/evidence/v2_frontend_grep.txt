export async function sendOtp(identifier, options = {}) {
  const normalized = normalizeIdentifier(identifier)
  if (!normalized) throw new Error('Email or phone is required.')

  const emailRedirectTo = resolveEmailRedirectTo(options.emailRedirectTo)

  if (isEmailIdentifier(normalized)) {
    // Try Cloudflare Worker first
    try {
      const result = await requestEmailOtpGateway(normalized.toLowerCase(), { emailRedirectTo })
      return result
    } catch (gatewayError) {
      if (!shouldFallbackToDirectEmailOtp(gatewayError)) {
        throw normalizeOtpAuthError(gatewayError, 'send')
      }
      // Fallback to Supabase direct
    }
    const { error } = await supabase.auth.signInWithOtp({
      email: normalized.toLowerCase(),
      options: { emailRedirectTo },
    })
    // ...
  }

  // Phone path: Supabase direct
  const phone = normalizePhone(normalized)
  const { error } = await supabase.auth.signInWithOtp({ phone })
  // ...
}
