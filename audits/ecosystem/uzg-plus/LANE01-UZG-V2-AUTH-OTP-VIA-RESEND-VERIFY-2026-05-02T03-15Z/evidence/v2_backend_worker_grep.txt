async function handleAuthOtpRequest(request, env) {
  const body = await parseRequestJson(request)
  const identifier = normalizeAuthOtpIdentifier(body?.identifier || body?.email)

  if (!identifier) return json(400, { code: 'AUTH_OTP_IDENTIFIER_REQUIRED', ... })
  if (!isAuthOtpEmailIdentifier(identifier)) return json(400, { code: 'AUTH_OTP_EMAIL_ONLY_GATEWAY', ... })

  const config = resolveSupabaseConfig(env)
  if (!config.ready) return json(503, { code: 'AUTH_OTP_GATEWAY_NOT_CONFIGURED', ... })

  const gatewayConfig = resolveAuthOtpEmailGatewayConfig(env)
  if (!gatewayConfig.ready) return json(503, { code: 'AUTH_EMAIL_PROVIDER_NOT_CONFIGURED', ... })

  // STEP 1: Generate OTP via Supabase admin API
  const generateResult = await generateAuthOtpEmailPayload(config, identifier.toLowerCase(), redirectTo)
  // → POST {SUPABASE_URL}/auth/v1/admin/generate_link
  // → Body: { type: 'magiclink', email, redirect_to }
  // → Response: { email_otp: '123456', action_link, redirect_to }

  if (!generateResult.ok) return json(..., { code: 'AUTH_OTP_GENERATE_FAILED', ... })

  // STEP 2: Deliver OTP via Resend
  const deliveryResult = await sendResendAuthOtpEmail(gatewayConfig, {
    email: identifier.toLowerCase(),
    otpCode: generateResult.otpCode,
  })

  if (!deliveryResult.ok) return json(..., { code: 'AUTH_EMAIL_PROVIDER_DELIVERY_FAILED', ... })

  return json(200, {
    ok: true,
    data: {
      sent: true,
      channel: 'email',
      email_hint: maskAuthOtpEmail(identifier),
      delivery_owner: 'worker-resend-email-gateway',
      verify_owner: 'supabase-auth-verifyOtp',
      cooldown_seconds: gatewayConfig.cooldownSeconds,
      requested_at: new Date().toISOString(),
    },
  })
}
